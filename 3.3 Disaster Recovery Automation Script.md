```powershell
# L3.3: Disaster Recovery Automation Script
# =============================================

function Get-FSMORoleStatus {
param([string]$Domain = (Get-ADDomain).DNSRoot)

Write-Host "=== FSMO Role Status ===" -ForegroundColor Cyan

try {
$FSMOInfo = netdom query fsmo /domain:$Domain 2>&1
Write-Host $FSMOInfo
return $FSMOInfo
}
catch {
Write-Host "7 Error: $_" -ForegroundColor Red
}
}

function Test-CriticalDCServices {
param([string]$DomainController)

Write-Host "`n=== Critical Services Check: $DomainController ===" `
-ForegroundColor Yellow

$CriticalServices = @(
"NTDS",      # Active Directory
"DNS",       # DNS Server
"Kdc",       # Kerberos
"NetLogon",  # Net Logon
"DFSR",      # DFS Replication
"EventLog"   # Event Log
)

foreach ($Service in $CriticalServices) {

try {

$ServiceStatus = Get-Service -Name $Service -ComputerName $DomainController `
-ErrorAction Stop

$Status = if ($ServiceStatus.Status -eq "Running") { "3" } else { "7" }

Write-Host "$Status $Service : $($ServiceStatus.Status)"
}
catch {
Write-Host "7 $Service : Cannot access"
}
}
}

function Invoke-FSMORoleSeizure {
param(
[string]$TargetDC,
[string]$Role = "All" # PDC, RID, Infra, Schema, DomainNaming, or All
)

Write-Host "=== FSMO Role Seizure ===" -ForegroundColor Red

Write-Host "WARNING: Use only when source DC is permanently unavailable!" `
-ForegroundColor Yellow

$Confirmation = Read-Host "Continue? (YES/NO)"

if ($Confirmation -ne "YES") {
Write-Host "Aborted." -ForegroundColor Yellow
return
}

try {

$ntdsutil = @"
activate instance ntds
roles
seize pdc
seize rid master
seize infrastructure master
seize schema master
seize domain naming master
quit
quit
"@

$ntdsutil | ntdsutil.exe

Write-Host "3 FSMO role seizure completed" -ForegroundColor Green

# Verify
Write-Host "`nVerifying new FSMO holders..."

Get-FSMORoleStatus
}
catch {
Write-Host "7 Error during seizure: $_" -ForegroundColor Red
}
}

function Invoke-DCFailover {
param(
[string[]]$FailedDCs,
[string[]]$HealthyDCs
)

Write-Host "=== DC Failover Procedure ===" -ForegroundColor Cyan

foreach ($FailedDC in $FailedDCs) {

Write-Host "`nProcessing failed DC: $FailedDC" -ForegroundColor Yellow

# Step 1: Remove failed DC from domain
Write-Host "[1] Attempting graceful removal..." -ForegroundColor Cyan

try {

$DC = Get-ADDomainController -Identity $FailedDC -ErrorAction Stop

Write-Host " Found: $($DC.HostName)"
}
catch {
Write-Host " 7 DC not accessible - metadata cleanup required"
}

# Step 2: Seize FSMO roles from healthy DC
Write-Host "[2] Seizing FSMO roles..." -ForegroundColor Cyan

if ($HealthyDCs.Count -gt 0) {

$TargetDC = $HealthyDCs[0]

Write-Host " Target: $TargetDC"

# Actual seizure happens here
}

# Step 3: Force replication
Write-Host "[3] Forcing replication on healthy DCs..." -ForegroundColor Cyan

foreach ($HealthyDC in $HealthyDCs) {

try {

Get-ADReplicationStatus -EnumerationServer $HealthyDC |
ForEach-Object { Sync-ADObject -Object $_.PartnerName }

Write-Host " 3 $HealthyDC synced"
}
catch {
Write-Host " n $HealthyDC : $_"
}
}

# Step 4: Metadata cleanup
Write-Host "[4] Metadata cleanup..." -ForegroundColor Cyan

Write-Host " Run: ntdsutil > activate instance ntds > metadata cleanup"
}

Write-Host "`n3 Failover procedure completed" -ForegroundColor Green
}

function Export-DCMetadata {
param(
[string]$FailedDC,
[string]$OutputPath = "C:\DR"
)

if (!(Test-Path $OutputPath)) {
New-Item -Path $OutputPath -ItemType Directory | Out-Null
}

Write-Host "Exporting metadata for $FailedDC..."

try {

$Metadata = @{
DCName = $FailedDC
ExportTime = Get-Date
FSMORoles = Get-FSMORoleStatus
ReplicationStatus = Get-ADReplicationStatus -EnumerationServer $FailedDC
DatabaseHealth = "Run dcdiag on DC before failure"
}

$Metadata | Export-Clixml -Path "$OutputPath\${FailedDC}_Metadata.xml"

Write-Host "3 Metadata exported to $OutputPath"
}
catch {
Write-Host "7 Error: $_" -ForegroundColor Red
}
}
```

# ⚙️ Functions Included

## 🏛️ 1. Retrieve FSMO Role Status

**Function:** `Get-FSMORoleStatus`

Displays the current holders of all Flexible Single Master Operations (FSMO) roles for the specified domain.

### Parameter

| Parameter | Description                                              |
| --------- | -------------------------------------------------------- |
| `Domain`  | Active Directory domain (defaults to the current domain) |

### Roles Displayed

* 👑 Schema Master
* 🌐 Domain Naming Master
* 🔑 PDC Emulator
* 🆔 RID Master
* 🏗️ Infrastructure Master

---

## 🩺 2. Verify Critical Domain Controller Services

**Function:** `Test-CriticalDCServices`

Checks whether essential Active Directory services are running on a specified Domain Controller.

### Parameter

| Parameter          | Description                   |
| ------------------ | ----------------------------- |
| `DomainController` | Domain Controller to validate |

### Services Checked

| Service  | Purpose                          |
| -------- | -------------------------------- |
| NTDS     | Active Directory Domain Services |
| DNS      | DNS Server                       |
| Kdc      | Kerberos Key Distribution Center |
| NetLogon | Authentication & Logon Services  |
| DFSR     | DFS Replication                  |
| EventLog | Windows Event Log                |

---

## ⚠️ 3. Perform FSMO Role Seizure

**Function:** `Invoke-FSMORoleSeizure`

Initiates an interactive FSMO role seizure using **NTDSUTIL**.

### Parameters

| Parameter  | Description                                        |
| ---------- | -------------------------------------------------- |
| `TargetDC` | Domain Controller that will receive the FSMO roles |
| `Role`     | Target role(s) to seize (default: All)             |

### Process

* Displays a warning message
* Requires manual confirmation
* Executes NTDSUTIL seizure commands
* Verifies the new FSMO role owners

---

## 🔄 4. Execute Domain Controller Failover

**Function:** `Invoke-DCFailover`

Provides a guided failover workflow for failed Domain Controllers.

### Parameters

| Parameter    | Description                    |
| ------------ | ------------------------------ |
| `FailedDCs`  | Failed Domain Controllers      |
| `HealthyDCs` | Operational Domain Controllers |

### Failover Workflow

1. Identify failed Domain Controllers
2. Attempt graceful removal
3. Prepare FSMO role seizure
4. Force replication
5. Perform metadata cleanup guidance

---

## 💾 5. Export Domain Controller Metadata

**Function:** `Export-DCMetadata`

Exports critical Domain Controller information for disaster recovery documentation.

### Parameters

| Parameter    | Description                              |
| ------------ | ---------------------------------------- |
| `FailedDC`   | Failed Domain Controller                 |
| `OutputPath` | Destination folder for exported metadata |

### Metadata Includes

* Domain Controller name
* Export timestamp
* FSMO role information
* Replication status
* Database health reminder

---

# 🚀 Usage Examples

### Display FSMO Role Owners

```powershell
Get-FSMORoleStatus
```

```powershell
Test-CriticalDCServices -DomainController "DC1"
```

### Initiate FSMO Role Seizure

```powershell
Invoke-FSMORoleSeizure -TargetDC "DC2" -Role "All"
```

### Execute Domain Controller Failover

```powershell
Invoke-DCFailover -FailedDCs "DC1" -HealthyDCs "DC2","DC3"
```

### Export Domain Controller Metadata

```powershell
Export-DCMetadata -FailedDC "DC1"
```

# ⚠️ Important Warning

> **FSMO role seizure is a destructive administrative operation.**
>
> Use **`Invoke-FSMORoleSeizure`** **only when the original FSMO role holder is permanently unavailable and cannot be recovered.** Performing a role seizure while the original Domain Controller may return to service can result in Active Directory inconsistencies.

> 💡 **Best Practice**
>
> Before executing disaster recovery operations:
>
> * Verify recent **System State** backups are available.
> * Confirm the failed Domain Controller cannot be recovered before seizing FSMO roles.
> * Replace placeholder values such as `DC1`, `DC2`, `DC3`, and `C:\DR` with values appropriate for your environment.
> * Validate Active Directory health (for example, with `dcdiag` and replication checks) before and after recovery activities.
