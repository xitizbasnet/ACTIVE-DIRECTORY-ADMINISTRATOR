# 📄 Script Code

```powershell id="gdh2x9"
# L2.3: Bulk User Password Reset Script
# ========================================

function Set-BulkUserPassword {
    param(
        [string]$CSVPath,
        [string]$LogPath = "C:\Logs\PasswordReset_$(Get-Date -f yyyyMMdd_HHmm).log"
    )

    # Validate input
    if (!(Test-Path $CSVPath)) {
        Write-Host "7 CSV file not found: $CSVPath" -ForegroundColor Red
        return
    }

    # Create log directory
    $LogDir = Split-Path -Path $LogPath

    if (!(Test-Path $LogDir)) {
        New-Item -Path $LogDir -ItemType Directory | Out-Null
    }

    $Users = Import-Csv -Path $CSVPath
    $Results = @()

    Write-Host "Processing $(($Users | Measure-Object).Count) users..." `
        -ForegroundColor Cyan

    foreach ($User in $Users) {

        $UserName = $User.SamAccountName
        $TempPassword = $User.TempPassword

        try {

            # Validate user exists
            $ADUser = Get-ADUser -Identity $UserName -ErrorAction Stop

            # Set password
            $SecurePassword = ConvertTo-SecureString $TempPassword -AsPlainText -Force

            Set-ADAccountPassword -Identity $UserName -NewPassword $SecurePassword `
                -Reset -ErrorAction Stop

            # Force password change at next logon
            Set-ADUser -Identity $UserName -ChangePasswordAtLogon $true `
                -ErrorAction Stop

            $Status = "SUCCESS"

            Write-Host "3 $UserName" -ForegroundColor Green
        }
        catch {

            $Status = "FAILED: $_"

            Write-Host "7 $UserName - $_" -ForegroundColor Red
        }

        # Log result
        $LogEntry = "[$(Get-Date -f 'yyyy-MM-dd HH:mm:ss')] $UserName : $Status"

        Add-Content -Path $LogPath -Value $LogEntry

        $Results += [PSCustomObject]@{
            UserName  = $UserName
            Status    = $Status
            Timestamp = Get-Date
        }
    }

    Write-Host "`n3 Log file: $LogPath" -ForegroundColor Green

    # Summary
    $Success = ($Results | Where-Object { $_.Status -eq "SUCCESS" }).Count

    Write-Host "`nSummary: $Success/$($Results.Count) successful" -ForegroundColor Cyan

    return $Results
}

function Generate-SecurePassword {
    param([int]$Length = 14)

    $Characters = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789!@#$%'

    $Password = -join (
        1..$Length |
        ForEach-Object {
            Get-Random -InputObject $Characters.ToCharArray()
        }
    )

    return $Password
}
```

# ⚙️ Functions Included

## 🔐 1. Bulk Password Reset

**Function:** `Set-BulkUserPassword`

Reads user accounts from a CSV file, resets passwords, forces password changes at next logon, and creates a detailed audit log.

### Parameters

| Parameter | Description                        |
| --------- | ---------------------------------- |
| `CSVPath` | Path to the input CSV file         |
| `LogPath` | Destination path for the audit log |

### Processing Steps

* ✅ Validates the CSV file exists
* 📂 Creates the log directory if needed
* 👤 Validates each Active Directory user
* 🔐 Resets user passwords
* 🔄 Forces password change at next logon
* 📝 Writes audit entries to a log file
* 📊 Returns a summary of successful and failed operations

## 🔑 2. Generate Secure Password

**Function:** `Generate-SecurePassword`

### Parameter

| Parameter | Description                                      |

| `Length`  | Desired password length (default: 14 characters) |
