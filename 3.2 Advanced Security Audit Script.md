# 📄 Script Code

```powershell
# L3.2: Advanced Security Audit Script
# ========================================

function Invoke-ADSecurityAudit {
param(
[string]$Domain = (Get-ADDomain).DNSRoot,
[string]$ReportPath = "C:\Reports"
)

Write-Host "=== AD Security Audit Report ===" -ForegroundColor Cyan

$Audit = @()

# 1. FIND INACTIVE ACCOUNTS
Write-Host "`n[1] Checking for inactive accounts..." -ForegroundColor Yellow

$InactiveUsers = Get-ADUser -Filter { LastLogonDate -lt (Get-Date).AddMonths(-3) } `
-Properties LastLogonDate | Select-Object Name, LastLogonDate

$Audit += [PSCustomObject]@{
Issue = "Inactive Users (>3 months)"
Count = $InactiveUsers.Count
Severity = if ($InactiveUsers.Count -gt 10) { "HIGH" } else { "MEDIUM" }
Details = "$($InactiveUsers.Count) accounts not logged in for 3+ months"
}

# 2. PRIVILEGED ACCOUNT REVIEW
Write-Host "[2] Auditing privileged accounts..." -ForegroundColor Yellow

$AdminAccounts = Get-ADGroupMember -Identity "Domain Admins" -Recursive
$EnterpriseAdmins = Get-ADGroupMember -Identity "Enterprise Admins" -Recursive

$PrivilegedCount = ($AdminAccounts | Measure-Object).Count + `
($EnterpriseAdmins | Measure-Object).Count

$Audit += [PSCustomObject]@{
Issue = "Privileged Account Count"
Count = $PrivilegedCount
Severity = if ($PrivilegedCount -gt 15) { "HIGH" } else { "INFO" }
Details = "Domain Admins: $($AdminAccounts.Count), Enterprise Admins: $($EnterpriseAdmins.Count)"
}

# 3. PASSWORD POLICY CHECK
Write-Host "[3] Checking password policies..." -ForegroundColor Yellow

$PasswordPolicy = Get-ADDefaultDomainPasswordPolicy
$PolicyIssues = @()

if ($PasswordPolicy.MinPasswordLength -lt 12) { $PolicyIssues += "Min length < 12" }
if ($PasswordPolicy.PasswordHistoryCount -lt 12) { $PolicyIssues += "History < 12" }
if ($PasswordPolicy.MaxPasswordAge -gt 90) { $PolicyIssues += "Max age > 90 days" }

$Audit += [PSCustomObject]@{
Issue = "Password Policy Compliance"
Count = $PolicyIssues.Count
Severity = if ($PolicyIssues.Count -gt 0) { "MEDIUM" } else { "OK" }
Details = if ($PolicyIssues.Count -eq 0) { "Compliant" } else { $PolicyIssues -join ", " }
}

# 4. SERVICE ACCOUNT ENUMERATION
Write-Host "[4] Auditing service accounts..." -ForegroundColor Yellow

$ServiceAccounts = Get-ADUser -Filter { UserAccountControl -band 0x80 } -Properties *

$UnchangedPasswords = $ServiceAccounts |
Where-Object { $_.PasswordLastSet -lt (Get-Date).AddMonths(-12) }

$Audit += [PSCustomObject]@{
Issue = "Service Accounts with Unchanged Passwords (>12 months)"
Count = $UnchangedPasswords.Count
Severity = if ($UnchangedPasswords.Count -gt 0) { "HIGH" } else { "OK" }
Details = "$($UnchangedPasswords.Count) accounts need password rotation"
}

# 5. KERBEROS AUDIT
Write-Host "[5] Checking Kerberos configuration..." -ForegroundColor Yellow

$SPNIssues = 0
$Users = Get-ADUser -Filter * -Properties ServicePrincipalName

foreach ($User in $Users) {
if ($User.ServicePrincipalName) { $SPNIssues++ }
}

$Audit += [PSCustomObject]@{
Issue = "Users with SPN (Kerberoasting Risk)"
Count = $SPNIssues
Severity = if ($SPNIssues -gt 5) { "HIGH" } else { "MEDIUM" }
Details = "$SPNIssues regular users have SPNs"
}

# Export report
if (!(Test-Path $ReportPath)) {
New-Item -Path $ReportPath -ItemType Directory | Out-Null
}

$ReportFile = "$ReportPath\Security_Audit_$(Get-Date -f yyyyMMdd_HHmm).csv"

$Audit | Export-Csv -Path $ReportFile -NoTypeInformation

Write-Host "`n3 Report saved: $ReportFile" -ForegroundColor Green

return $Audit
}

function Get-ADUsersWithWeakPasswords {
param([string]$Domain = (Get-ADDomain).DNSRoot)

Write-Host "`n=== Weak Password Detection ===" -ForegroundColor Yellow

# Check for users with passwords not changed recently
Get-ADUser -Filter * -Properties PasswordLastSet, PasswordNotRequired `
-Server $Domain | Where-Object {
$_.PasswordNotRequired -eq $true -or
$_.PasswordLastSet -lt (Get-Date).AddMonths(-6)
} | Select-Object Name, PasswordNotRequired, PasswordLastSet
}
```

# ⚙️ Functions Included

## 🛡️ 1. Active Directory Security Audit

**Function:** `Invoke-ADSecurityAudit`

Performs a multi-stage security assessment of the Active Directory environment and exports the results to a CSV report.

### Parameters

| Parameter    | Description                                                     |
| ------------ | --------------------------------------------------------------- |
| `Domain`     | Target Active Directory domain (defaults to the current domain) |
| `ReportPath` | Directory where the security audit report will be saved         |

### Security Checks Performed

#### 🔍 1. Inactive User Accounts

Identifies user accounts that have not logged in for more than **3 months**.

Severity is determined by the number of inactive accounts detected.

---

#### 👑 2. Privileged Account Review

Audits membership of:

* Domain Admins
* Enterprise Admins

Calculates the total number of privileged accounts.

---

#### 🔐 3. Password Policy Compliance

Evaluates the default domain password policy, including:

* Minimum password length
* Password history
* Maximum password age

Reports compliance issues where applicable.

---

#### ⚙️ 4. Service Account Audit

Identifies service accounts whose passwords have not been changed for over **12 months**.

---

#### 🎫 5. Kerberos Configuration Review

Counts user accounts configured with Service Principal Names (SPNs), highlighting potential **Kerberoasting** exposure.

---

## 🔑 2. Weak Password Detection

**Function:** `Get-ADUsersWithWeakPasswords`

Lists accounts where:

* Passwords are **not required**
* Passwords have not been changed within the past **6 months**

### Parameter

| Parameter | Description                    |
| --------- | ------------------------------ |
| `Domain`  | Target Active Directory domain |

The audit evaluates the following security areas:

| Category                    | Assessment                                |
| --------------------------- | ----------------------------------------- |
| 👤 Inactive Users           | Accounts inactive for more than 3 months  |
| 👑 Privileged Accounts      | Domain Admins and Enterprise Admins       |
| 🔐 Password Policy          | Complexity, history, and expiration       |
| ⚙️ Service Accounts         | Password age exceeding 12 months          |
| 🎫 Kerberos                 | Users with Service Principal Names (SPNs) |
| 🔑 Weak Password Indicators | PasswordNotRequired and stale passwords   |
