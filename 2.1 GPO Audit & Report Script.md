# 📄 Script Code

```powershell
# L2.1: GPO Audit & Report Script
# ==================================

function Get-GPOAuditReport {
    param(
        [string]$Domain = (Get-ADDomain).DNSRoot,
        [string]$OutputPath = "C:\Reports"
    )

    if (!(Test-Path $OutputPath)) {
        New-Item -Path $OutputPath -ItemType Directory
    }

    $Report = @()
    $GPOs = Get-GPO -All -Domain $Domain

    Write-Host "Auditing $($GPOs.Count) GPOs..." -ForegroundColor Cyan

    foreach ($GPO in $GPOs) {
        try {
            # Get GPO status
            $GPODetails = $GPO | Get-GPOReport -ReportType Xml

            # Get linked OUs
            $LinkedOUs = @(
                Get-ADOrganizationalUnit -Filter * |
                Where-Object { ($_ | Get-GPLink).GpoId -contains $GPO.Id }
            )

            # Get security group filtering
            $Permissions = $GPO | Select-Object -ExpandProperty GpoStatus

            $Report += [PSCustomObject]@{
                GPOName      = $GPO.DisplayName
                Status       = $GPO.GpoStatus
                LinkedOUs    = $LinkedOUs.Count
                Owner        = $GPO.Owner
                CreatedDate  = $GPO.CreationTime
                ModifiedDate = $GPO.ModificationTime
            }
        }
        catch {
            Write-Host "n Error processing $($GPO.DisplayName): $_" -ForegroundColor Yellow
        }
    }

    # Export report
    $ReportFile = "$OutputPath\GPO_Audit_$(Get-Date -Format 'yyyyMMdd_HHmm').csv"

    $Report | Export-Csv -Path $ReportFile -NoTypeInformation

    Write-Host "`n3 Report generated: $ReportFile" -ForegroundColor Green

    return $Report
}

function Get-DisabledGPOs {
    param([string]$Domain = (Get-ADDomain).DNSRoot)

    Write-Host "`n=== Disabled GPOs ===" -ForegroundColor Yellow

    Get-GPO -All -Domain $Domain |
    Where-Object { $_.GpoStatus -eq "AllSettingsDisabled" } |
    Select-Object DisplayName, Owner, ModificationTime
}

function Find-UnlinkedGPOs {
    param([string]$Domain = (Get-ADDomain).DNSRoot)

    Write-Host "`n=== Unlinked GPOs (Potential Cleanup) ===" -ForegroundColor Yellow

    $AllGPOs = Get-GPO -All -Domain $Domain

    $AllGPOs | Where-Object {
        @(
            Get-ADOrganizationalUnit -Filter * |
            Where-Object { ($_ | Get-GPLink).GpoId -contains $_.Id }
        ).Count -eq 0
    } |
    Select-Object DisplayName, Owner, ModificationTime
}
```

# ⚙️ Functions Included

## 📊 1. Generate GPO Audit Report

**Function:** `Get-GPOAuditReport`

Performs a complete audit of all Group Policy Objects in the specified domain and exports the results to a CSV report.

### Parameters

| Parameter    | Description                                                 |
| ------------ | ----------------------------------------------------------- |
| `Domain`     | Target Active Directory domain (defaults to current domain) |
| `OutputPath` | Directory where the report will be saved                    |

### Report Includes

* 📄 GPO Name
* ⚙️ GPO Status
* 📁 Number of Linked Organizational Units (OUs)
* 👤 Owner
* 📅 Creation Date
* 🕒 Last Modification Date

---

## 🚫 2. List Disabled GPOs

**Function:** `Get-DisabledGPOs`

Identifies Group Policy Objects with all settings disabled.

### Parameter

| Parameter | Description                    |
| --------- | ------------------------------ |
| `Domain`  | Target Active Directory domain |

---

## 🧹 3. Find Unlinked GPOs

**Function:** `Find-UnlinkedGPOs`

Searches for Group Policy Objects that are not linked to any Organizational Unit and may be candidates for cleanup.

### Parameter

| Parameter | Description                    |
| --------- | ------------------------------ |
| `Domain`  | Target Active Directory domain |
